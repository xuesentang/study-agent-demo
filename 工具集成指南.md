```python
"""
工具基类定义
所有工具必须继承此类
"""
from abc import ABC, abstractmethod
from typing import Dict, Any, List, Optional
from pydantic import BaseModel


class ToolParameter(BaseModel):
    """工具参数定义"""
    name: str
    type: str
    description: str
    required: bool = True
    default: Any = None


class ToolResult(BaseModel):
    """工具执行结果"""
    success: bool
    data: Any
    error: Optional[str] = None


class BaseTool(ABC):
    """
    工具基类
    
    所有工具必须继承此类并实现以下方法
    """
    
    # 工具元数据（必须定义）
    name: str = ""  # 工具名称，唯一标识
    description: str = ""  # 工具功能描述
    parameters: List[ToolParameter] = []  # 参数定义
    
    @abstractmethod
    async def execute(self, **kwargs) -> ToolResult:
        """
        执行工具
        
        Args:
            **kwargs: 工具参数
            
        Returns:
            ToolResult: 执行结果
        """
        pass
    
    def get_schema(self) -> Dict[str, Any]:
        """
        获取工具Schema（用于LLM理解）
        
        Returns:
            符合OpenAI Function Calling格式的Schema
        """
        return {
            "type": "function",
            "function": {
                "name": self.name,
                "description": self.description,
                "parameters": {
                    "type": "object",
                    "properties": {
                        param.name: {
                            "type": param.type,
                            "description": param.description
                        }
                        for param in self.parameters
                    },
                    "required": [
                        param.name for param in self.parameters 
                        if param.required
                    ]
                }
            }
        }


class ToolRegistry:
    """
    工具注册中心
    
    管理所有可用工具，提供注册和获取功能
    """
    
    _tools: Dict[str, BaseTool] = {}
    
    @classmethod
    def register(cls, tool: BaseTool) -> None:
        """
        注册工具
        
        Args:
            tool: 工具实例
            
        Raises:
            ValueError: 工具名称重复时抛出
        """
        if tool.name in cls._tools:
            raise ValueError(f"工具 '{tool.name}' 已存在")
        cls._tools[tool.name] = tool
        print(f"✅ 工具已注册: {tool.name}")
    
    @classmethod
    def get(cls, name: str) -> Optional[BaseTool]:
        """
        获取工具
        
        Args:
            name: 工具名称
            
        Returns:
            工具实例或None
        """
        return cls._tools.get(name)
    
    @classmethod
    def get_all_tools(cls) -> List[BaseTool]:
        """
        获取所有工具
        
        Returns:
            工具实例列表
        """
        return list(cls._tools.values())
    
    @classmethod
    def get_all_schemas(cls) -> List[Dict[str, Any]]:
        """
        获取所有工具的Schema
        
        Returns:
            Schema列表（用于LLM）
        """
        return [tool.get_schema() for tool in cls._tools.values()]
    
    @classmethod
    def list_tools(cls) -> List[str]:
        """
        列出所有工具名称
        
        Returns:
            工具名称列表
        """
        return list(cls._tools.keys())


# 自动导入并注册所有工具的函数
def auto_register_tools():
    """
    自动发现并注册所有工具
    
    在应用启动时调用
    """
    import importlib
    import pkgutil
    from backend.tools import __path__ as tools_path
    
    # 遍历 tools 包下的所有模块
    for _, module_name, _ in pkgutil.iter_modules(tools_path):
        if module_name == 'base':
            continue
        try:
            module = importlib.import_module(f'backend.tools.{module_name}')
            # 查找模块中的工具类并自动注册
            for attr_name in dir(module):
                attr = getattr(module, attr_name)
                if (isinstance(attr, type) and 
                    issubclass(attr, BaseTool) and 
                    attr != BaseTool and
                    hasattr(attr, 'name') and attr.name):
                    ToolRegistry.register(attr())
        except Exception as e:
            print(f"⚠️ 加载工具模块 {module_name} 失败: {e}")
```

```python
"""
文件操作工具
支持：读取、写入、列出目录、创建目录、删除文件
"""
import os
import aiofiles
from pathlib import Path
from typing import List, Optional
from backend.tools.base import BaseTool, ToolParameter, ToolResult


class FileTool(BaseTool):
    """文件操作工具"""
    
    name = "file_operation"
    description = "执行文件操作：读取、写入、列出目录、创建目录、删除文件或目录"
    parameters = [
        ToolParameter(
            name="action",
            type="string",
            description="操作类型: read(读取), write(写入), list(列出目录), mkdir(创建目录), delete(删除), exists(检查存在)"
        ),
        ToolParameter(
            name="path",
            type="string",
            description="文件或目录路径（相对路径基于项目根目录）"
        ),
        ToolParameter(
            name="content",
            type="string",
            description="写入内容（仅write操作需要）",
            required=False,
            default=""
        ),
        ToolParameter(
            name="encoding",
            type="string",
            description="文件编码，默认utf-8",
            required=False,
            default="utf-8"
        )
    ]
    
    # 允许操作的根目录（安全限制）
    ALLOWED_ROOTS = [
        "documents",      # 文档目录
        "downloads",      # 下载目录
        "output",         # 输出目录
        "temp",           # 临时目录
        "backend/data",   # 后端数据
    ]
    
    def _validate_path(self, path: str) -> tuple[bool, str]:
        """
        验证路径安全性
        
        Args:
            path: 用户输入的路径
            
        Returns:
            (是否合法, 错误信息或规范化路径)
        """
        # 规范化路径
        path = os.path.normpath(path)
        
        # 禁止绝对路径和向上跳转
        if os.path.isabs(path) or '..' in path.split(os.sep):
            return False, "路径不能是绝对路径或包含'..'"
        
        # 检查是否在允许目录下
        parts = path.split(os.sep)
        if not parts or parts[0] not in self.ALLOWED_ROOTS:
            allowed = ', '.join(self.ALLOWED_ROOTS)
            return False, f"只能在以下目录操作: {allowed}"
        
        # 确保目录存在
        root_dir = parts[0]
        if not os.path.exists(root_dir):
            os.makedirs(root_dir, exist_ok=True)
        
        return True, path
    
    async def execute(self, **kwargs) -> ToolResult:
        """执行文件操作"""
        action = kwargs.get('action', '').lower()
        path = kwargs.get('path', '')
        content = kwargs.get('content', '')
        encoding = kwargs.get('encoding', 'utf-8')
        
        # 验证路径
        is_valid, result = self._validate_path(path)
        if not is_valid:
            return ToolResult(success=False, data=None, error=result)
        path = result
        
        try:
            if action == 'read':
                return await self._read_file(path, encoding)
            elif action == 'write':
                return await self._write_file(path, content, encoding)
            elif action == 'list':
                return await self._list_dir(path)
            elif action == 'mkdir':
                return await self._make_dir(path)
            elif action == 'delete':
                return await self._delete(path)
            elif action == 'exists':
                return await self._exists(path)
            else:
                return ToolResult(
                    success=False, 
                    data=None, 
                    error=f"未知操作: {action}"
                )
        except Exception as e:
            return ToolResult(success=False, data=None, error=str(e))
    
    async def _read_file(self, path: str, encoding: str) -> ToolResult:
        """读取文件"""
        if not os.path.exists(path):
            return ToolResult(success=False, data=None, error=f"文件不存在: {path}")
        if not os.path.isfile(path):
            return ToolResult(success=False, data=None, error=f"不是文件: {path}")
        
        async with aiofiles.open(path, 'r', encoding=encoding) as f:
            content = await f.read()
        
        return ToolResult(success=True, data={
            "path": path,
            "content": content,
            "size": len(content),
            "encoding": encoding
        })
    
    async def _write_file(self, path: str, content: str, encoding: str) -> ToolResult:
        """写入文件"""
        # 确保父目录存在
        parent = os.path.dirname(path)
        if parent:
            os.makedirs(parent, exist_ok=True)
        
        async with aiofiles.open(path, 'w', encoding=encoding) as f:
            await f.write(content)
        
        return ToolResult(success=True, data={
            "path": path,
            "bytes_written": len(content.encode(encoding)),
            "message": f"文件已保存: {path}"
        })
    
    async def _list_dir(self, path: str) -> ToolResult:
        """列出目录内容"""
        if not os.path.exists(path):
            return ToolResult(success=False, data=None, error=f"目录不存在: {path}")
        
        items = []
        for item in os.listdir(path):
            item_path = os.path.join(path, item)
            items.append({
                "name": item,
                "type": "directory" if os.path.isdir(item_path) else "file",
                "size": os.path.getsize(item_path) if os.path.isfile(item_path) else None
            })
        
        return ToolResult(success=True, data={
            "path": path,
            "items": items,
            "count": len(items)
        })
    
    async def _make_dir(self, path: str) -> ToolResult:
        """创建目录"""
        os.makedirs(path, exist_ok=True)
        return ToolResult(success=True, data={
            "path": path,
            "message": f"目录已创建: {path}"
        })
    
    async def _delete(self, path: str) -> ToolResult:
        """删除文件或目录"""
        import shutil
        
        if not os.path.exists(path):
            return ToolResult(success=False, data=None, error=f"路径不存在: {path}")
        
        if os.path.isdir(path):
            shutil.rmtree(path)
            msg = f"目录已删除: {path}"
        else:
            os.remove(path)
            msg = f"文件已删除: {path}"
        
        return ToolResult(success=True, data={
            "path": path,
            "message": msg
        })
    
    async def _exists(self, path: str) -> ToolResult:
        """检查文件是否存在"""
        exists = os.path.exists(path)
        is_file = os.path.isfile(path) if exists else False
        is_dir = os.path.isdir(path) if exists else False
        
        return ToolResult(success=True, data={
            "path": path,
            "exists": exists,
            "is_file": is_file,
            "is_directory": is_dir
        })
```

```python
"""
联网搜索工具
支持：百度搜索、Bing搜索（需要API Key）
"""
import os
import json
import httpx
from typing import List, Dict, Any, Optional
from backend.tools.base import BaseTool, ToolParameter, ToolResult


class SearchTool(BaseTool):
    """联网搜索工具"""
    
    name = "web_search"
    description = "执行联网搜索，获取实时信息"
    parameters = [
        ToolParameter(
            name="query",
            type="string",
            description="搜索关键词"
        ),
        ToolParameter(
            name="engine",
            type="string",
            description="搜索引擎: bing(默认), baidu",
            required=False,
            default="bing"
        ),
        ToolParameter(
            name="num_results",
            type="integer",
            description="返回结果数量(1-10)",
            required=False,
            default=5
        )
    ]
    
    async def execute(self, **kwargs) -> ToolResult:
        """执行搜索"""
        query = kwargs.get('query', '')
        engine = kwargs.get('engine', 'bing').lower()
        num_results = min(max(kwargs.get('num_results', 5), 1), 10)
        
        if not query:
            return ToolResult(success=False, data=None, error="搜索关键词不能为空")
        
        try:
            if engine == 'bing':
                return await self._search_bing(query, num_results)
            elif engine == 'baidu':
                return await self._search_baidu(query, num_results)
            else:
                return ToolResult(
                    success=False, 
                    data=None, 
                    error=f"不支持的搜索引擎: {engine}"
                )
        except Exception as e:
            return ToolResult(success=False, data=None, error=f"搜索失败: {str(e)}")
    
    async def _search_bing(self, query: str, num: int) -> ToolResult:
        """Bing搜索（需要Azure订阅Key）"""
        api_key = os.getenv("BING_SEARCH_API_KEY")
        if not api_key:
            return ToolResult(
                success=False, 
                data=None, 
                error="未配置BING_SEARCH_API_KEY环境变量"
            )
        
        endpoint = "https://api.bing.microsoft.com/v7.0/search"
        headers = {"Ocp-Apim-Subscription-Key": api_key}
        params = {
            "q": query,
            "count": num,
            "textDecorations": False,
            "textFormat": "HTML"
        }
        
        async with httpx.AsyncClient() as client:
            response = await client.get(endpoint, headers=headers, params=params)
            response.raise_for_status()
            data = response.json()
        
        results = []
        for item in data.get("webPages", {}).get("value", [])[:num]:
            results.append({
                "title": item.get("name", ""),
                "url": item.get("url", ""),
                "snippet": item.get("snippet", "")
            })
        
        return ToolResult(success=True, data={
            "query": query,
            "engine": "bing",
            "total_results": data.get("webPages", {}).get("totalEstimatedMatches", 0),
            "results": results
        })
    
    async def _search_baidu(self, query: str, num: int) -> ToolResult:
        """
        百度搜索（使用第三方服务或爬虫）
        这里使用 SerpAPI 作为示例
        """
        api_key = os.getenv("SERPAPI_KEY")
        if not api_key:
            return ToolResult(
                success=False, 
                data=None, 
                error="未配置SERPAPI_KEY环境变量（百度搜索需要）"
            )
        
        endpoint = "https://serpapi.com/search"
        params = {
            "engine": "baidu",
            "q": query,
            "num": num,
            "api_key": api_key
        }
        
        async with httpx.AsyncClient() as client:
            response = await client.get(endpoint, params=params)
            response.raise_for_status()
            data = response.json()
        
        results = []
        for item in data.get("organic_results", [])[:num]:
            results.append({
                "title": item.get("title", ""),
                "url": item.get("link", ""),
                "snippet": item.get("snippet", "")
            })
        
        return ToolResult(success=True, data={
            "query": query,
            "engine": "baidu",
            "results": results
        })
```

```python
"""
网页抓取工具
支持：获取网页内容、提取正文、获取链接
"""
import re
import httpx
from typing import List, Dict, Any, Optional
from urllib.parse import urljoin, urlparse
from backend.tools.base import BaseTool, ToolParameter, ToolResult


class WebScrapeTool(BaseTool):
    """网页抓取工具"""
    
    name = "web_scrape"
    description = "抓取网页内容，提取文本、链接等信息"
    parameters = [
        ToolParameter(
            name="url",
            type="string",
            description="要抓取的网页URL"
        ),
        ToolParameter(
            name="action",
            type="string",
            description="操作类型: content(获取正文), links(获取链接), html(获取原始HTML)",
            required=False,
            default="content"
        ),
        ToolParameter(
            name="timeout",
            type="integer",
            description="超时时间(秒)",
            required=False,
            default=10
        )
    ]
    
    async def execute(self, **kwargs) -> ToolResult:
        """执行网页抓取"""
        url = kwargs.get('url', '')
        action = kwargs.get('action', 'content').lower()
        timeout = kwargs.get('timeout', 10)
        
        if not url:
            return ToolResult(success=False, data=None, error="URL不能为空")
        
        # 验证URL格式
        if not url.startswith(('http://', 'https://')):
            url = 'https://' + url
        
        try:
            async with httpx.AsyncClient(timeout=timeout, follow_redirects=True) as client:
                headers = {
                    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.0"
                }
                response = await client.get(url, headers=headers)
                response.raise_for_status()
                html = response.text
            
            if action == 'html':
                return ToolResult(success=True, data={
                    "url": url,
                    "content": html[:50000],  # 限制大小
                    "length": len(html)
                })
            elif action == 'links':
                return await self._extract_links(html, url)
            else:  # content
                return await self._extract_content(html, url)
                
        except httpx.TimeoutException:
            return ToolResult(success=False, data=None, error="请求超时")
        except httpx.HTTPStatusError as e:
            return ToolResult(success=False, data=None, error=f"HTTP错误: {e.response.status_code}")
        except Exception as e:
            return ToolResult(success=False, data=None, error=f"抓取失败: {str(e)}")
    
    async def _extract_content(self, html: str, url: str) -> ToolResult:
        """提取正文内容（简单实现）"""
        # 移除script和style标签
        html = re.sub(r'<script[^>]*>.*?</script>', '', html, flags=re.DOTALL | re.IGNORECASE)
        html = re.sub(r'<style[^>]*>.*?</style>', '', html, flags=re.DOTALL | re.IGNORECASE)
        
        # 提取title
        title_match = re.search(r'<title[^>]*>(.*?)</title>', html, re.IGNORECASE | re.DOTALL)
        title = title_match.group(1).strip() if title_match else ""
        
        # 提取正文（简单策略：找p标签和div标签中的文本）
        # 更复杂的可以使用BeautifulSoup或readability-lxml
        text_parts = []
        
        # 提取p标签内容
        for match in re.finditer(r'<p[^>]*>(.*?)</p>', html, re.DOTALL | re.IGNORECASE):
            text = self._clean_html(match.group(1))
            if len(text) > 20:  # 过滤短文本
                text_parts.append(text)
        
        # 如果没有p标签，尝试提取div中的文本
        if not text_parts:
            # 移除所有标签，保留文本
            text = re.sub(r'<[^>]+>', ' ', html)
            text = self._clean_html(text)
            text_parts = [text]
        
        content = '\n\n'.join(text_parts[:20])  # 限制段落数
        content = content[:10000]  # 限制总长度
        
        return ToolResult(success=True, data={
            "url": url,
            "title": title,
            "content": content,
            "paragraphs": len(text_parts)
        })
    
    async def _extract_links(self, html: str, base_url: str) -> ToolResult:
        """提取页面中的所有链接"""
        links = []
        seen = set()
        
        for match in re.finditer(r'href=["\'](.*?)["\']', html, re.IGNORECASE):
            href = match.group(1)
            
            # 跳过锚点和javascript
            if href.startswith(('#', 'javascript:', 'mailto:', 'tel:')):
                continue
            
            # 转换为绝对URL
            full_url = urljoin(base_url, href)
            
            # 去重
            if full_url in seen:
                continue
            seen.add(full_url)
            
            # 提取链接文本
            # 简单实现：找a标签中的文本
            link_pattern = rf'<a[^>]*href=["\']{re.escape(href)}["\'][^>]*>(.*?)</a>'
            text_match = re.search(link_pattern, html, re.IGNORECASE | re.DOTALL)
            text = self._clean_html(text_match.group(1)) if text_match else ""
            
            links.append({
                "url": full_url,
                "text": text[:100]  # 限制长度
            })
        
        return ToolResult(success=True, data={
            "url": base_url,
            "total_links": len(links),
            "links": links[:50]  # 限制返回数量
        })
    
    def _clean_html(self, text: str) -> str:
        """清理HTML标签和多余空白"""
        # 移除剩余标签
        text = re.sub(r'<[^>]+>', ' ', text)
        # 解码HTML实体
        text = text.replace('&nbsp;', ' ')
        text = text.replace('&lt;', '<')
        text = text.replace('&gt;', '>')
        text = text.replace('&amp;', '&')
        text = text.replace('&quot;', '"')
        # 合并空白
        text = ' '.join(text.split())
        return text.strip()
```

```python
            default=""
```
```python
        ),
        ToolParameter(

            type="string",
            description="保存目录（相对路径，默认downloads）",
            required=False,
            default="downloads"
        ),
        ToolParameter(
            name="timeout",
            type="integer",
            description="超时时间(秒)",
            required=False,
            default=60
        )
    ]
    
    # 允许下载的文件类型
    ALLOWED_EXTENSIONS = [
        '.txt', '.pdf', '.doc', '.docx', '.xls', '.xlsx',
        '.ppt', '.pptx', '.csv', '.json', '.xml', '.md',
        '.jpg', '.jpeg', '.png', '.gif', '.bmp', '.svg',
        '.mp3', '.mp4', '.wav', '.avi', '.mov',
        '.zip', '.rar', '.7z', '.tar', '.gz',
        '.py', '.js', '.html', '.css', '.ts', '.tsx',
        '.java', '.cpp', '.c', '.h', '.go', '.rs'
    ]
    
    # 最大文件大小（100MB）
    MAX_FILE_SIZE = 100 * 1024 * 1024
    
    async def execute(self, **kwargs) -> ToolResult:
        """执行下载"""
        url = kwargs.get('url', '')
        filename = kwargs.get('filename', '')
        save_dir = kwargs.get('save_dir', 'downloads')
        timeout = kwargs.get('timeout', 60)
        
        if not url:
            return ToolResult(success=False, data=None, error="URL不能为空")
        
        # 验证URL
        if not url.startswith(('http://', 'https://')):
            url = 'https://' + url
        
        parsed = urlparse(url)
        if not parsed.netloc:
            return ToolResult(success=False, data=None, error="无效的URL")
        
        # 确定文件名
        if not filename:
            filename = os.path.basename(parsed.path) or 'download'
        
        # 验证文件扩展名
        ext = os.path.splitext(filename)[1].lower()
        if ext not in self.ALLOWED_EXTENSIONS:
            return ToolResult(
                success=False, 
                data=None, 
                error=f"不支持的文件类型: {ext}。允许的类型: {', '.join(self.ALLOWED_EXTENSIONS[:10])}..."
            )
        
        # 确保保存目录存在
        save_dir = os.path.normpath(save_dir)
        if '..' in save_dir.split(os.sep):
            return ToolResult(success=False, data=None, error="保存目录不能包含'..'")
        
        os.makedirs(save_dir, exist_ok=True)
        filepath = os.path.join(save_dir, filename)
        
        # 检查文件是否已存在
        if os.path.exists(filepath):
            # 添加数字后缀
            base, ext = os.path.splitext(filename)
            counter = 1
            while os.path.exists(filepath):
                filename = f"{base}_{counter}{ext}"
                filepath = os.path.join(save_dir, filename)
                counter += 1
        
        try:
            # 下载文件
            downloaded_size = 0
            md5_hash = hashlib.md5()
            
            async with httpx.AsyncClient(timeout=timeout, follow_redirects=True) as client:
                headers = {
                    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.0"
                }
                
                async with client.stream('GET', url, headers=headers) as response:
                    response.raise_for_status()
                    
                    # 检查文件大小
                    content_length = response.headers.get('content-length')
                    if content_length:
                        total_size = int(content_length)
                        if total_size > self.MAX_FILE_SIZE:
                            return ToolResult(
                                success=False, 
                                data=None, 
                                error=f"文件太大({total_size}字节)，最大允许{self.MAX_FILE_SIZE}字节"
                            )
                    
                    # 写入文件
                    async with aiofiles.open(filepath, 'wb') as f:
                        async for chunk in response.aiter_bytes(chunk_size=8192):
                            downloaded_size += len(chunk)
                            
                            # 检查大小限制
                            if downloaded_size > self.MAX_FILE_SIZE:
                                os.remove(filepath)
                                return ToolResult(
                                    success=False, 
                                    data=None, 
                                    error=f"下载文件超过大小限制"
                                )
                            
                            await f.write(chunk)
                            md5_hash.update(chunk)
            
            return ToolResult(success=True, data={
                "url": url,
                "filename": filename,
                "filepath": filepath,
                "size": downloaded_size,
                "md5": md5_hash.hexdigest(),
                "message": f"文件下载成功: {filepath}"
            })
            
        except httpx.TimeoutException:
            if os.path.exists(filepath):
                os.remove(filepath)
            return ToolResult(success=False, data=None, error="下载超时")
        except Exception as e:
            if os.path.exists(filepath):
                os.remove(filepath)
            return ToolResult(success=False, data=None, error=f"下载失败: {str(e)}")
```

        try:
            # 尝试使用 fpdf2（推荐）
            try:
                from fpdf import FPDF
                return await self._generate_with_fpdf(content, title, filepath, fmt)

    async def _generate_with_fpdf(self, content: str, title: str, filepath: str, fmt: str) -> ToolResult:
        """使用 fpdf2 生成PDF"""
        from fpdf import FPDF
        
        class PDF(FPDF):
            def header(self):
                self.set_font('DejaVu', 'B', 12)
                self.cell(0, 10, title, 0, 1, 'C')
                self.ln(5)
            
            def footer(self):
                self.set_y(-15)
                self.set_font('DejaVu', 'I', 8)
                self.cell(0, 10, f'Page {self.page_no()}', 0, 0, 'C')
        
        pdf = PDF()
        
        # 添加Unicode字体支持
        font_paths = [
            'backend/fonts/DejaVuSans.ttf',
            '/usr/share/fonts/truetype/dejavu/DejaVuSans.ttf',
            'C:/Windows/Fonts/arial.ttf',
        ]
        
        font_added = False
        for font_path in font_paths:
            if os.path.exists(font_path):
                pdf.add_font('DejaVu', '', font_path, uni=True)
                pdf.add_font('DejaVu', 'B', font_path, uni=True)
                pdf.add_font('DejaVu', 'I', font_path, uni=True)
                font_added = True
                break
        
        if not font_added:
            # 使用默认字体（仅支持ASCII）
            pdf.set_font('Arial', '', 12)
        else:
            pdf.set_font('DejaVu', '', 12)
        
        pdf.add_page()
        
        # 处理内容
        if fmt == 'markdown':
            lines = self._markdown_to_text(content)
        else:
            lines = content.split('\n')
        
        for line in lines:
            if line.startswith('# '):
                pdf.set_font('DejaVu' if font_added else 'Arial', 'B', 16)
                pdf.cell(0, 10, line[2:], 0, 1)
                pdf.set_font('DejaVu' if font_added else 'Arial', '', 12)
            elif line.startswith('## '):
                pdf.set_font('DejaVu' if font_added else 'Arial', 'B', 14)
                pdf.cell(0, 8, line[3:], 0, 1)
                pdf.set_font('DejaVu' if font_added else 'Arial', '', 12)
            elif line.startswith('### '):
                pdf.set_font('DejaVu' if font_added else 'Arial', 'B', 12)
                pdf.cell(0, 6, line[4:], 0, 1)
                pdf.set_font('DejaVu' if font_added else 'Arial', '', 12)
            elif line.strip() == '':
                pdf.ln(5)
            else:
                pdf.multi_cell(0, 6, line)
        
        pdf.output(filepath)

    async def _generate_with_reportlab(self, content: str, title: str, filepath: str, fmt: str) -> ToolResult:
        """使用 reportlab 生成PDF"""
        from reportlab.lib.pagesizes import A4
        from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer
        from reportlab.lib.styles import getSampleStyleSheet
        from reportlab.lib.units import inch
        
        doc = SimpleDocTemplate(filepath, pagesize=A4)
        styles = getSampleStyleSheet()
        story = []
        
        # 添加标题
        story.append(Paragraph(title, styles['Title']))
        story.append(Spacer(1, 0.2 * inch))
        
        # 处理内容
        if fmt == 'markdown':
            lines = self._markdown_to_text(content)
        else:
            lines = content.split('\n')
        
        for line in lines:
            if line.startswith('# '):
                story.append(Paragraph(line[2:], styles['Heading1']))
            elif line.startswith('## '):
                story.append(Paragraph(line[3:], styles['Heading2']))
            elif line.startswith('### '):
                story.append(Paragraph(line[4:], styles['Heading3']))
            elif line.strip() == '':
                story.append(Spacer(1, 0.1 * inch))
            else:
                story.append(Paragraph(line, styles['BodyText']))
        
        doc.build(story)
        
        return ToolResult(success=True, data={
            "filepath": filepath,
            "filename": os.path.basename(filepath),
            "message": f"PDF已生成: {filepath}"
        })
    
    async def _generate_with_weasyprint(self, content: str, title: str, filepath: str, fmt: str) -> ToolResult:
        """使用 weasyprint 生成PDF（支持HTML）"""
        from weasyprint import HTML, CSS
        
        if fmt == 'markdown':
            import markdown
            html_content = markdown.markdown(content)
        elif fmt == 'html':
            html_content = content
        else:
            html_content = f"<pre>{content}</pre>"
        
        full_html = f"""
        <!DOCTYPE html>
        <html>
        <head>
            <meta charset="UTF-8">
            <title>{title}</title>
            <style>
                body {{ font-family: Arial, sans-serif; margin: 40px; }}
                h1 {{ color: #333; }}
                h2 {{ color: #555; }}
                p {{ line-height: 1.6; }}
            </style>
        </head>
        <body>
            <h1>{title}</h1>
            {html_content}
        </body>
        </html>
        """
        
        HTML(string=full_html).write_pdf(filepath)
        
        return ToolResult(success=True, data={
            "filepath": filepath,
            "filename": os.path.basename(filepath),
            "message": f"PDF已生成: {filepath}"
        })
    
    def _markdown_to_text(self, markdown_text: str) -> list:
        """简单Markdown解析为文本行"""
        lines = markdown_text.split('\n')
        result = []
        
        for line in lines:
            # 移除Markdown格式标记
            line = line.strip()
            if line.startswith('```'):
                continue  # 跳过代码块标记
            # 保留标题标记
            result.append(line)
        
        return result

```python
            DownloadTool(),
            PDFTool()
        ]
        
        for tool in tools_to_register:
            try:
                ToolRegistry.register(tool)
            except ValueError as e:
                print(f"⚠️ {e}")
        
        print(f"✅ 共注册 {len(ToolRegistry.list_tools())} 个工具")
    
    def get_available_tools(self) -> List[Dict[str, Any]]:
        """获取所有可用工具的Schema"""
        return ToolRegistry.get_all_schemas()
    
    async def execute_tool(self, tool_name: str, parameters: Dict[str, Any]) -> ToolResult:
        """
        执行指定工具
        
        Args:
            tool_name: 工具名称
            parameters: 工具参数
            
        Returns:
            ToolResult: 执行结果
        """
        tool = ToolRegistry.get(tool_name)
        if not tool:
            return ToolResult(
                success=False, 
                data=None, 
                error=f"工具不存在: {tool_name}"
            )
        
        try:
            result = await tool.execute(**parameters)
            return result
        except Exception as e:
            return ToolResult(
                success=False, 
                data=None, 
                error=f"工具执行异常: {str(e)}"
            )
    
    async def chat_with_tools(
        self, 
        messages: List[Dict[str, str]], 
        enable_tools: bool = True
    ) -> Dict[str, Any]:
        """
        带工具调用的对话
        
        Args:
            messages: 对话历史
            enable_tools: 是否启用工具调用
            
        Returns:
            包含回复和工具调用结果的字典
        """
        tools = self.get_available_tools() if enable_tools else None
        
        # 构建系统提示词
        system_prompt = """你是一位专业的概率论与数理统计辅导老师，同时也是一个智能助手。

你可以使用以下工具来帮助用户：
1. file_operation - 文件操作（读写、列出目录等）
2. web_search - 联网搜索实时信息
3. web_scrape - 抓取网页内容
4. terminal - 执行终端命令（受限）
5. download - 下载网络资源
6. generate_pdf - 生成PDF文档

当用户需要这些功能时，请使用工具调用来完成。"""
        
        # 构建完整消息
        full_messages = [{"role": "system", "content": system_prompt}] + messages
        
        # 调用LLM
        if tools:
            response = await self.client.chat.completions.create(
                model=self.model,
                messages=full_messages,
                tools=tools,
                tool_choice="auto",
                temperature=0.7,
                max_tokens=2000
            )
        else:
            response = await self.client.chat.completions.create(
                model=self.model,
                messages=full_messages,
                temperature=0.7,
                max_tokens=2000
            )
        
        message = response.choices[0].message
        
        # 检查是否有工具调用
        if message.tool_calls:
            tool_results = []
            
            for tool_call in message.tool_calls:
                tool_name = tool_call.function.name
                try:
                    parameters = json.loads(tool_call.function.arguments)
                except json.JSONDecodeError:
                    parameters = {}
                
                # 执行工具
                result = await self.execute_tool(tool_name, parameters)
                tool_results.append({
                    "tool": tool_name,
                    "parameters": parameters,
                    "result": result
                })
            
            # 将工具结果返回给LLM生成最终回复
            tool_messages = full_messages + [
                {
                    "role": "assistant",
                    "content": message.content or "",
                    "tool_calls": [
                        {
                            "id": tc.id,
                            "type": "function",
                            "function": {
                                "name": tc.function.name,
                                "arguments": tc.function.arguments
                            }
                        }
                        for tc in message.tool_calls
                    ]
                }
            ]
            
            # 添加工具结果
            for i, tr in enumerate(tool_results):
                tool_messages.append({
                    "role": "tool",
                    "tool_call_id": message.tool_calls[i].id,
                    "content": json.dumps({
                        "success": tr["result"].success,
                        "data": tr["result"].data,
                        "error": tr["result"].error
                    }, ensure_ascii=False)
                })
            
            # 再次调用LLM生成最终回复
            final_response = await self.client.chat.completions.create(
                model=self.model,
                messages=tool_messages,
                temperature=0.7,
                max_tokens=2000
            )
            
            return {
                "content": final_response.choices[0].message.content,
                "tool_calls": tool_results,
                "model": self.model
            }
        
        # 普通回复（无工具调用）
        return {
            "content": message.content,
            "tool_calls": [],
            "model": self.model
        }
```

## 九、扩展开发

如需添加新工具：

1. **创建工具文件**：`backend/tools/my_tool.py`
2. **继承BaseTool**：实现 `execute` 方法
3. **定义元数据**：`name`, `description`, `parameters`
4. **注册工具**：在 `__init__.py` 中导入
5. **重启服务**：自动加载新工具

示例：

```python
# backend/tools/my_tool.py
from backend.tools.base import BaseTool, ToolParameter, ToolResult

class MyTool(BaseTool):
    name = "my_tool"
    description = "我的自定义工具"
    parameters = [
        ToolParameter(name="param1", type="string", description="参数1")
    ]
    
    async def execute(self, **kwargs) -> ToolResult:
        param1 = kwargs.get('param1', '')
        # 实现逻辑
        return ToolResult(success=True, data={"result": param1})
```
